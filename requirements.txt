# ComfyUI-ChatterboxBangla — requirements
# =========================================
# Install into ComfyUI's venv:
#   source /path/to/ComfyUI/.venv/bin/activate
#   pip install -r requirements.txt
#
# NOTE: We do NOT install chatterbox-tts from pip because it requires
# transformers==5.2.0, which conflicts with qwen-tts (transformers==4.57.3).
# Instead, we git-clone the chatterbox-finetuning repo at runtime and
# import directly from it. The node handles this automatically.
#
# Prerequisites (already in ComfyUI venv, no action needed):
#   torch, torchaudio, numpy, huggingface_hub
# =========================================
